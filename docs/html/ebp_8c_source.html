252int ebp_register_op(uint32_t op_id, ebp_op_t fn)
253{
254 int i;
255
256 if (!fn) {
257 EBA_ERR("eba_register_op: null function pointer\n");
258 return -EINVAL;
259 }
260
261 /* Check if this op_id is already registered */
262 for (i = 0; i < MAX_OP_COUNT; i++) {
263 if (op_entries[i].op_id == op_id) {
264 EBA_ERR("eba_register_op: op_id %u already exists (slot %d)\n", op_id, i);
265 return -EEXIST;
266 }
267 }
268
269 /* Find a free entry */
270 for (i = 0; i < MAX_OP_COUNT; i++) {
271 if (op_entries[i].op_ptr == NULL && op_entries[i].op_id == 0) {
272 op_entries[i].op_id = op_id;
273 op_entries[i].op_ptr = fn;
274 EBA_INFO("Registered op_id %u in slot %d\n", op_id, i);
275 return 0;
276 }
277 }
278 EBA_ERR("eba_register_op: No space for new op_id %u\n", op_id);
279 return -ENOSPC;
