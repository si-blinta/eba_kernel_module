32 /* IOCTL handler: dispatches commands to allocation, read, and write operations */
33 static long eba_ioctl(struct file *file, unsigned int cmd, unsigned long arg)
34 {
35 int ret = 0;
36 struct eba_alloc_data alloc_data;
37 struct eba_rw_data rw_data;
38 struct eba_remote_alloc remote_alloc;
39 struct eba_remote_write remote_write;
40 struct eba_remote_read remote_read;
41 void *kbuf; /* Temporary kernel buffer for data transfers */
42
43 switch (cmd) {
44 case EBA_IOCTL_ALLOC:
45 /* Copy the allocation parameters from user space */
46 if (copy_from_user(&alloc_data, (void __user *)arg, sizeof(alloc_data)))
47 return -EFAULT;
48
49 alloc_data.buff_id = (uint64_t)eba_internals_malloc(alloc_data.size,
50 alloc_data.life_time);
51 if (!alloc_data.buff_id)
52 ret = -ENOMEM;
53
54 /* Return the allocation info to user space */
55 if (copy_to_user((void __user *)arg, &alloc_data, sizeof(alloc_data)))
56 ret = -EFAULT;
57 break;
58
59 case EBA_IOCTL_WRITE:
60 if (copy_from_user(&rw_data, (void __user *)arg, sizeof(rw_data)))
61 return -EFAULT;
62 kbuf = kmalloc(rw_data.size, GFP_KERNEL);
63 if (!kbuf)
64 return -ENOMEM;
65 if (copy_from_user(kbuf, (void __user *)(rw_data.user_addr), rw_data.size)) {
66 kfree(kbuf);
67 return -EFAULT;
68 }
69 ret = eba_internals_write(kbuf, rw_data.buff_id, rw_data.off, rw_data.size);
70 kfree(kbuf);
71 break;
72
73 case EBA_IOCTL_READ:
74 if (copy_from_user(&rw_data, (void __user *)arg, sizeof(rw_data)))
75 return -EFAULT;
76 kbuf = kmalloc(rw_data.size, GFP_KERNEL);
77 if (!kbuf)
78 return -ENOMEM;
79 ret = eba_internals_read(kbuf, rw_data.buff_id, rw_data.off, rw_data.size);
80 if (ret) {
81 kfree(kbuf);
82 return ret;
83 }
84 if (copy_to_user((void __user *)(rw_data.user_addr), kbuf, rw_data.size))
85 ret = -EFAULT;
86 kfree(kbuf);
87 break;
88
89 case EBA_IOCTL_REMOTE_ALLOC:
90 if (copy_from_user(&remote_alloc, (void __user *)arg, sizeof(remote_alloc)))
91 return -EFAULT;
92 ret = ebp_remote_alloc(remote_alloc.size,remote_alloc.life_time,remote_alloc.buffer_id,remote_alloc.mac);
93 break;
94
95 case EBA_IOCTL_REMOTE_WRITE:
96 if (copy_from_user(&remote_write, (void __user *)arg, sizeof(remote_write)))
97 return -EFAULT;
98 ret = ebp_remote_write(remote_write.buff_id,remote_write.offset,remote_write.size,remote_write.payload,remote_write.mac);
99 break;
100 case EBA_IOCTL_REMOTE_READ:
101 if (copy_from_user(&remote_read, (void __user *)arg, sizeof(remote_read)))
102 return -EFAULT;
103 ret = ebp_remote_read(remote_read.dst_buffer_id,remote_read.src_buffer_id,remote_read.dst_offset,remote_read.src_offset,remote_read.size,remote_read.mac);
104 break;
105
106 default:
107 ret = -ENOTTY;
108 break;
109 }
110 return ret;
111 }

145 /* Module initialization: register character device and initialize memory pool */
146 static int __init eba_module_init(void)
147 {
148 int ret;
149 /* Allocate a character device region */
150 ret = alloc_chrdev_region(&eba_devno, 0, 1, "eba");
151 if (ret < 0)
152 goto err_alloc;
153
154 /* Initialize and add the character device */
155 cdev_init(&eba_cdev, &eba_fops);
156 eba_cdev.owner = THIS_MODULE;
157 ret = cdev_add(&eba_cdev, eba_devno, 1);
158 if (ret)
159 goto err_cdev;
160
161 /* Create a device class using the single-argument version of class_create() */
162 eba_class = class_create("eba_class");
163 if (IS_ERR(eba_class)) {
164 ret = PTR_ERR(eba_class);
165 goto err_class;
166 }
167 /* Create the device node /dev/eba */
168 device_create(eba_class, NULL, eba_devno, NULL, "eba");
169
170 /* Initialize the memory pool */
171 ret = eba_internals_mempool_init();
172 if (ret) {
173 EBA_ERR("Mempool Init failed\n");
174 goto err_mempool;
175 }
176 // Initialize the timer:
177 timer_setup(&eba_timer, eba_timer_callback, 0);
178 // Schedule the timer for the first time
179 mod_timer(&eba_timer, jiffies + msecs_to_jiffies(EBA_CLEAN_BUFFER_CALLBACK_TIMER));
180
181 ebp_init();
182 EBA_INFO("Module loaded\n");
183 return 0;
184
185 err_mempool:
186 device_destroy(eba_class, eba_devno);
187 class_destroy(eba_class);
188 err_class:
189 cdev_del(&eba_cdev);
190 err_cdev:
191 unregister_chrdev_region(eba_devno, 1);
192 err_alloc:
193 return ret;
194 }
195
196 /* Module cleanup: free memory pool and unregister character device */
197 static void __exit eba_module_exit(void)
198 {
199 del_timer_sync(&eba_timer);
200 eba_internals_mempool_free();
201 device_destroy(eba_class, eba_devno);
202 class_destroy(eba_class);
203 cdev_del(&eba_cdev);
204 unregister_chrdev_region(eba_devno, 1);
205 ebp_exit();
206 EBA_INFO("Module unloaded\n");
207 }
208 module_init(eba_module_init);
209 module_exit(eba_module_exit);
