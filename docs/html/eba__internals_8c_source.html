22int eba_internals_mempool_init(void)
23{
24 unsigned long align = 8; /* Set allocation alignment to 8 bytes */
25 unsigned long phys_addr; /* Dummy physical address for vmalloc'd memory */
26
27 /* Create the generic memory pool.
28 * The second parameter (-1) specifies flags; here, no special flags are needed.
29 */
30 eba_pool = gen_pool_create(align, -1);
31 if (!eba_pool) {
32 EBA_ERR("EBA: Failed to create memory pool\n");
33 return 1;
34 }
35
36 /* Allocate a contiguous block of virtual memory using vmalloc.
37 * This method supports very large allocations (subject to system limits) but
38 * the returned memory is not physically contiguous.
39 */
40 eba_pool_mem = vmalloc(MEMORY_POOL_SIZE);
41 if (!eba_pool_mem) {
42 EBA_ERR("EBA: Failed to allocate memory for the pool\n");
43 gen_pool_destroy(eba_pool);
44 return 1;
45 }
46
47 /* Since vmalloc'd memory is not physically contiguous,
48 * we cannot reliably compute a physical base address.
49 * We pass a dummy value (e.g., 0) to gen_pool_add_virt().
50 */
51 phys_addr = 0;
52 gen_pool_add_virt(eba_pool, (unsigned long)eba_pool_mem, phys_addr, MEMORY_POOL_SIZE, -1);
53
54 /* Initialize the spinlock to protect the global tracking list */
55 spin_lock_init(&eba_buffer_list_lock);
56
57 EBA_INFO("EBA: Memory pool initialized successfully\n");
58 return 0;
59}

78
79 /* Check if the memory pool is initialized */

106

110 else

113 INIT_LIST_HEAD(&buf->node);
114
115 /* Add the tracking structure to the global list in a thread-safe manner */
116 spin_lock(&eba_buffer_list_lock);

118 spin_unlock(&eba_buffer_list_lock);
119
120 EBA_DBG("EBA: Allocated memory at %llu, size: %llu bytes, lifetime: %llu sec\n",

137int eba_internals_free(void *ptr)
138{
139 struct eba_buffer *buf;
140 int found = 0;
141
142 /* Check if the memory pool is initialized and if the pointer is valid */
143 if (!eba_pool || !ptr)
144 return -1;
145
146 /* Lock the global list before searching for the tracking structure */
147 spin_lock(&eba_buffer_list_lock);
148 list_for_each_entry(buf, &eba_buffer_list, node) {
149 if (buf->address == (unsigned long)ptr) {
150 list_del(&buf->node);
151 found = 1;
152 break;
153 }
154 }
155 spin_unlock(&eba_buffer_list_lock);
156
157 /* If no tracking structure is found, log an error and return -1 */
158 if (!found) {
159 EBA_ERR("EBA: Attempt to free unknown memory at %llu\n", (uint64_t)ptr);
160 return -1;
161 }
162
163 /* Free the memory from the gen_pool using the stored size */
164 gen_pool_free(eba_pool, (unsigned long)ptr, buf->size);
165
166 /* Free the tracking structure */
167 kfree(buf);
168
169 EBA_DBG("EBA: Freed memory at %llu, size: %llu bytes\n", (uint64_t)ptr, buf->size);
170 return 0;

179void eba_internals_mempool_free(void)
180{
181 struct eba_buffer *buf, *tmp;
182 int outstanding = 0;
183
184 // Count outstanding allocations
185 spin_lock(&eba_buffer_list_lock);
186 list_for_each_entry(buf, &eba_buffer_list, node) {
187 outstanding++;
188 }
189 EBA_DBG("EBA: %d outstanding buffers before cleanup\n", outstanding);
190 if (outstanding > 0) {
191 list_for_each_entry_safe(buf, tmp, &eba_buffer_list, node) {
192 EBA_DBG("EBA: Forcing free of buffer at %llu, size: %llu bytes\n",
193 buf->address, buf->size);
194 list_del(&buf->node);
195 gen_pool_free(eba_pool, buf->address, buf->size);
196 kfree(buf);
197 }
198 }
199
200 // Now destroy the gen_pool and free the backing memory
201 if (eba_pool) {
202 gen_pool_destroy(eba_pool);
203 eba_pool = NULL;
204 }
205 if (eba_pool_mem) {
206 vfree(eba_pool_mem);
207 eba_pool_mem = NULL;
208 }
209 spin_unlock(&eba_buffer_list_lock);
210 EBA_INFO("EBA: Memory pool freed successfully\n");
211}

364/*
365 * eba_internals_rw_stress_test - Performs a stress test for read/write operations.
366 *
367 * For each allocation in the test, this function:
368 * 1. Allocates a buffer using eba_internals_malloc() with a given size and lifetime.
369 * 2. Creates a test pattern.
370 * 3. Uses eba_internals_write() to write the pattern into the allocated buffer.
371 * 4. Reads back the data using eba_internals_read() into a temporary buffer.
372 * 5. Compares the read data with the original pattern to verify correctness.
373 * 6. Frees the allocated buffer with eba_internals_free().
374 *
375 * The test is performed for a predetermined number of iterations.
376 */
377int eba_internals_rw_stress_test(void)
378{
379 const int num_allocs = 100; /* Number of buffers to allocate for the test */
380 void **allocations; /* Array to store allocated buffer pointers */
381 int *sizes; /* Array to store the sizes for each buffer */
382 int i, ret = 0;
383 unsigned char *pattern = NULL;
384 unsigned char *read_back = NULL;
385
386 /* Allocate arrays to track allocated buffers and their sizes */
387 allocations = kmalloc_array(num_allocs, sizeof(void *), GFP_KERNEL);
388 if (!allocations)
389 return -ENOMEM;
390
391 sizes = kmalloc_array(num_allocs, sizeof(int), GFP_KERNEL);
392 if (!sizes) {
393 kfree(allocations);
394 return -ENOMEM;
395 }
396
397 /* Initialize the sizes and set each allocation pointer to NULL */
398 for (i = 0; i < num_allocs; i++) {
399 sizes[i] = 16 + ((i * 31) % 1024); /* Buffer size will vary between 16 and about 1040 bytes */
400 allocations[i] = NULL;
401 }
402
403 /* Allocation and test phase */
404 for (i = 0; i < num_allocs; i++) {
405 /* Allocate a buffer with a lifetime of 10 seconds */
406 allocations[i] = eba_internals_malloc(sizes[i], 10);
407 if (!allocations[i]) {
408 EBA_ERR("EBA-RW-STRESS: Allocation %d of size %d failed\n", i, sizes[i]);
409 ret = -ENOMEM;
410 goto cleanup;
411 }
412
413 /* Allocate temporary buffers for writing and reading */
414 pattern = kmalloc(sizes[i], GFP_KERNEL);
415 if (!pattern) {
416 ret = -ENOMEM;
417 goto cleanup;
418 }
419 read_back = kmalloc(sizes[i], GFP_KERNEL);
420 if (!read_back) {
421 kfree(pattern);
422 ret = -ENOMEM;
423 goto cleanup;
424 }
425
426 /*
427 * Fill the pattern buffer with a test pattern.
428 * For example, store a sequence of incrementing values modulo 256.
429 */
430 {
431 int j;
432 for (j = 0; j < sizes[i]; j++)
433 pattern[j] = (unsigned char)(j & 0xFF);
434 }
435
436 /* Write the test pattern into the allocated buffer */
437 ret = eba_internals_write(pattern, (uint64_t)allocations[i], 0, sizes[i]);
438 if (ret != 0) {
439 EBA_ERR("EBA-RW-STRESS: Write failed on buffer %d\n", i);
440 kfree(pattern);
441 kfree(read_back);
442 goto cleanup;
443 }
444
445 /* Read back the data into the read_back buffer */
446 ret = eba_internals_read(read_back, (uint64_t)allocations[i], 0, sizes[i]);
447 if (ret != 0) {
448 EBA_ERR("EBA-RW-STRESS: Read failed on buffer %d\n", i);
449 kfree(pattern);
450 kfree(read_back);
451 goto cleanup;
452 }
453
454 /* Compare the written pattern with the read-back data */
455 if (memcmp(pattern, read_back, sizes[i]) != 0) {
456 EBA_ERR("EBA-RW-STRESS: Data mismatch on buffer %d\n", i);
457 ret = -EINVAL;
458 kfree(pattern);
459 kfree(read_back);
460 goto cleanup;
461 }
462
463 /* Free the temporary buffers */
464 kfree(pattern);
465 kfree(read_back);
466
467 EBA_INFO("EBA-RW-STRESS: Buffer %d (size %d) verified successfully\n", i, sizes[i]);
468 }
469
470 EBA_INFO("EBA-RW-STRESS: All read/write operations completed successfully\n");
471 ret = 0;
472
473cleanup:
474 /* Free all allocated buffers */
475 for (i = 0; i < num_allocs; i++) {
476 if (allocations[i])
477 eba_internals_free(allocations[i]);
478 }
479 kfree(allocations);
480 kfree(sizes);
481 return ret;
482}
