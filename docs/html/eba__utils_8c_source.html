23int eba_utils_file_to_buf(const char *filepath, uint64_t buff_id)
24{
25 struct file *filp;
26 loff_t pos = 0;
27 loff_t file_size;
28 char *kbuf = NULL;
29 ssize_t bytes_read;
30 int ret = 1; /* default to error */
31
32 /* Open the file read-only */
33 filp = filp_open(filepath, O_RDONLY, 0);
34 if (IS_ERR(filp)) {
35 EBA_ERR("Failed to open file %s, err=%ld\n", filepath, PTR_ERR(filp));
36 return 1;
37 }
38
39 /* Get file size via i_size_read() */
40 file_size = i_size_read(file_inode(filp));
41 if (file_size <= 0) {
42 EBA_ERR("File %s is empty or error reading size\n", filepath);
43 goto out_close;
44 }
45
46 /* Safety check if extremely large */
47 if (file_size > ULONG_MAX) {
48 EBA_ERR("File %s is too large (%lld bytes)\n", filepath, file_size);
49 goto out_close;
50 }
51
52 /* Allocate a kernel buffer to read the file contents into */
53 kbuf = kmalloc(file_size, GFP_KERNEL);
54 if (!kbuf) {
55 EBA_ERR("Failed to allocate temp buffer\n");
56 goto out_close;
57 }
58
59 /* Read the entire file into kbuf */
60 bytes_read = kernel_read(filp, kbuf, file_size, &pos);
61 if (bytes_read < 0 || bytes_read != file_size) {
62 EBA_ERR("Failed to read entire file %s, bytes_read=%zd\n",
63 filepath, bytes_read);
64 goto out_free;
65 }
66
67 /* Write that data into the EBA buffer buff_id at offset=0 */
68 if (eba_internals_write(kbuf, buff_id, 0, file_size) < 0) {
69 EBA_ERR("eba_internals_write() failed for file %s\n", filepath);
70 goto out_free;
71 }
72
73 EBA_INFO("Read %lld bytes from file %s into EBA buffer 0x%llx\n",
74 file_size, filepath, buff_id);
75 ret = 0; /* success */
76
77out_free:
78 kfree(kbuf);
79out_close:
80 filp_close(filp, NULL);
81 return ret;
82}

92int eba_utils_buf_to_file(uint64_t buff_id, uint64_t size, const char *filepath)
93{
94 struct file *filp;
95 loff_t pos = 0;
96 char *kbuf = NULL;
97 ssize_t bytes_written;
98 int ret;
99
100 if (size == 0) {
101 EBA_ERR("eba_utils_buf_to_file: zero size, nothing to write\n");
102 return -EINVAL;
103 }
104
105 /* Allocate a kernel buffer of 'size' to read from EBA */
106 kbuf = kmalloc(size, GFP_KERNEL);
107 if (!kbuf)
108 return -ENOMEM;
109
110 /* Read from EBA buffer into kbuf */
111 ret = eba_internals_read(kbuf, buff_id, 0, size);
112 if (ret < 0) {
113 EBA_ERR("eba_internals_read() failed on buff_id 0x%llx\n", buff_id);
114 goto out_free;
115 }
116
117 /* Open or create the file (truncate if exists) */
118 filp = filp_open(filepath, O_WRONLY | O_CREAT | O_TRUNC, 0600);
119 if (IS_ERR(filp)) {
120 EBA_ERR("Failed to open file %s for writing, err=%ld\n",
121 filepath, PTR_ERR(filp));
122 ret = PTR_ERR(filp);
123 filp = NULL;
124 goto out_free;
125 }
126
127 /* Write kbuf to file */
128 bytes_written = kernel_write(filp, kbuf, size, &pos);
129 if (bytes_written < 0 || bytes_written != size) {
130 EBA_ERR("Failed to write entire buffer to file %s, bytes_written=%zd\n",
131 filepath, bytes_written);
132 if (bytes_written >= 0)
133 ret = -EIO;
134 else
135 ret = bytes_written;
136 goto out_close;
137 }
138
139 EBA_INFO("Wrote %llu bytes from EBA buffer 0x%llx to file %s\n",
140 size, buff_id, filepath);
141 ret = 0; /* success */
142
143out_close:
144 if (filp)
145 filp_close(filp, NULL);
146out_free:
147 kfree(kbuf);
148 return ret;
149}

154int eba_export_node_specs(void)
155{
156 int i;
157 char filepath[256];
158
159 EBA_INFO("Exporting node_specs buffers to files...\n");
160
161 for (i = 0; i < MAX_NODE_COUNT; i++) {
162 if (node_infos[i].id != -1 && node_infos[i].node_specs != 0) {
163 struct eba_buffer *buf;
164 int found = 0;
165 uint64_t buff_id = node_infos[i].node_specs;
166
167 /* Acquire global lock, search the buffer list for buff_id */
168 spin_lock(&eba_buffer_list_lock);
169 list_for_each_entry(buf, &eba_buffer_list, node) {
170 if (buf->address == buff_id) {
171 found = 1;
172 break;
173 }
174 }
175 spin_unlock(&eba_buffer_list_lock);
176
177 if (!found) {
178 EBA_ERR("Node %d has node_specs=0x%llx, but no matching buffer\n",
179 i, buff_id);
180 continue;
181 }
182
183 /* Build the path like "/var/lib/eba/node_2_specs" */
184 snprintf(filepath, sizeof(filepath),
185 "/var/lib/eba/node_%d_specs", i);
186
187 /* Dump the entire buffer to that file */
188 eba_utils_buf_to_file(buff_id, buf->size, filepath);
189 }
190 }
191
192 return 0;
193}
