3int send_raw_ethernet_packet(const unsigned char *payload, size_t payload_len,const unsigned char *dst_mac,int protocol,const char *ifname)
4{
5 struct net_device *dev = NULL;
6 struct sk_buff *skb = NULL;
7 struct ethhdr *eth;
8 int ret;
9
10 /* Look up the network device by name */
11 dev = dev_get_by_name(&init_net, ifname);
12 if (!dev) {
13 EBA_ERR("send_raw_ethernet_packet: Could not find device %s\n", ifname);
14 return -ENODEV;
15 }
16 /* Calculate total length: Ethernet header + payload */
17 size_t packet_len = ETH_HLEN + payload_len;
18
19 /*
20 * Allocate an sk_buff.
21 */
22 skb = alloc_skb(packet_len + dev->needed_tailroom, GFP_KERNEL);
23 if (!skb) {
24 EBA_ERR("send_raw_ethernet_packet: Failed to allocate sk_buff\n");
25 dev_put(dev);
26 return -ENOMEM;
27 }
28
29 /* Reserve headroom for the Ethernet header */
30 skb_reserve(skb, ETH_HLEN);
31
32 /* Set the device for the packet */
33 skb->dev = dev;
34 skb->protocol = htons(protocol);
35 skb->pkt_type = PACKET_OUTGOING;
36
37 /* Build the Ethernet header */
38 eth = (struct ethhdr *)skb_push(skb, ETH_HLEN);
39 /* Fill in destination MAC address (provided as argument) */
40 memcpy(eth->h_dest, dst_mac, ETH_ALEN);
41 /* Use the device's own MAC address as source */
42 memcpy(eth->h_source, dev->dev_addr, ETH_ALEN);
43 /* Set the protocol field */
44 eth->h_proto = htons(protocol);
45
46 /* Append payload data to the packet */
47 if (payload && payload_len > 0) {
48 void *payload_ptr = skb_put(skb, payload_len);
49 memcpy(payload_ptr, payload, payload_len);
50 }
51 /* Queue the packet for transmission */
52 ret = dev_queue_xmit(skb);
53 if (ret < 0) {
54 EBA_ERR("send_raw_ethernet_packet: Packet transmission failed: %d\n", ret);
55 /* In case of error, the kernel should free the skb */
56 } else {
57 EBA_INFO("send_raw_ethernet_packet: Packet transmitted successfully on %s\n", ifname);
58 }
59 /* Release the reference to the network device */
60 dev_put(dev);
61 return ret;
62}

88int eba_net_set_mtu(const char *ifname, int new_mtu)
89{
90 struct net_device *dev;
91 int ret = 0;
92
93 if (!ifname)
94 return -EINVAL;
95
96 /* Look up the network device by name */
97 dev = dev_get_by_name(&init_net, ifname);
98 if (!dev) {
99 EBA_ERR("EBA_NET: Could not find device %s\n", ifname);
100 return -ENODEV;
101 }
102
103 /* Check if the device's driver provides the ndo_change_mtu callback */
104 if (dev->netdev_ops && dev->netdev_ops->ndo_change_mtu) {
105 ret = dev->netdev_ops->ndo_change_mtu(dev, new_mtu);
106 if (ret) {
107 EBA_ERR("EBA_NET: Failed to change MTU on %s to %d, error: %d\n",
108 dev->name, new_mtu, ret);
109 } else {
110 EBA_INFO("EBA_NET: Successfully changed MTU on %s to %d\n",
111 dev->name, new_mtu);
112 }
113 } else {
114 EBA_ERR("EBA_NET: Device %s does not support MTU change\n", dev->name);
115 ret = -EOPNOTSUPP;
116 }
117
118 dev_put(dev);
119 return ret;

146char *build_invoke_req_packet(uint32_t iid, uint32_t opid,
147 const char *args, uint64_t args_len,
148 const char *payload, uint64_t payload_len,
149 uint64_t *out_len)
150{
151 uint64_t total_len = sizeof(struct ebp_invoke_req) + args_len + payload_len;
152 char *buf = kmalloc(total_len, GFP_KERNEL);
153 if (!buf)
154 return NULL;
155
156 struct ebp_invoke_req *req = (struct ebp_invoke_req *)buf;
157 req->header.msgType = EBP_MSG_INVOKE;
158 req->iid = htonl(iid);
159 req->opid = htonl(opid);
160 /* Set args_len as the sum of the write header length and the payload length */
161 req->args_len = cpu_to_be64(args_len + payload_len);
162 if (args && args_len > 0)
163 memcpy(buf + sizeof(struct ebp_invoke_req), args, args_len);
164 if (payload && payload_len > 0)
165 memcpy(buf + sizeof(struct ebp_invoke_req) + args_len, payload, payload_len);
166 *out_len = total_len;
167 return buf;
168}
